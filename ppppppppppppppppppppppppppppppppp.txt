            canvas.height = window.innerHeight;
            
            // Matrix characters
            const matrixChars = "ABCDEFGHIJKLMNOPQRSTUVWXYZ0123456789$#@%&[]";
            const fontSize = 14;
            const columns = canvas.width / fontSize;
            
            // Create an array of drops - one per column
            const drops = [];
            for (let i = 0; i < columns; i++) {
                drops[i] = Math.floor(Math.random() * canvas.height / fontSize);
            }
            
            // Draw the matrix animation
            function drawMatrix() {
                // Semi-transparent black to create trail effect
                ctx.fillStyle = 'rgba(0, 0, 0, 0.04)';
                ctx.fillRect(0, 0, canvas.width, canvas.height);
                
                ctx.fillStyle = '#0f0';
                ctx.font = `${fontSize}px monospace`;
                
                for (let i = 0; i < drops.length; i++) {
                    // Random character
                    const text = matrixChars.charAt(Math.floor(Math.random() * matrixChars.length));
                    
                    // x coordinate: i * fontSize, y coordinate: drops[i] * fontSize
                    ctx.fillText(text, i * fontSize, drops[i] * fontSize);
                    
                    // Reset drop if it reached bottom or randomly
                    if (drops[i] * fontSize > canvas.height && Math.random() > 0.975) {
                        drops[i] = 0;
                    }
                    
                    // Move drop down
                    drops[i]++;
                }
            }

                const line = document.createElement('div');
                line.className = 'code-line';

                    line.innerHTML = `<span class="prompt">${text.split(' ')[0]}</span> <span class="ip-address">${text.split(' ').slice(1).join(' ')}</span>`;
                } else if (type === "access") {
                    line.innerHTML = `<span class="prompt">${text.split(' ')[0]}</span> <span class="access-granted">${text.split(' ').slice(1).join(' ')}</span>`;
                } else {
                    line.innerHTML = `<span class="prompt">${text.split(' ')[0]}</span> ${text.split(' ').slice(1).join(' ')}`;
                }
                
                panel.appendChild(line);
                panel.scrollTop = panel.scrollHeight;
            }
            
            // Function to update loading progress
            function updateProgress() {
                progress += Math.random() * 8;
                if (progress > 100) progress = 100;
                
                loadingBar.style.width = `${progress}%`;
                percentage.textContent = `${Math.floor(progress)}%`;
